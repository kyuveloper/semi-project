## 막내팀 세미 프로젝트 DB 1차

````
CREATE TABLE IF NOT EXISTS USER_TABLE
(
    USER_CODE INT AUTO_INCREMENT COMMENT '유저 코드',
    USER_NICKNAME VARCHAR(10) NOT NULL COMMENT '닉네임',
    USER_EMAIL VARCHAR(30) NOT NULL COMMENT '유저 이메일',
    USER_PASSWORD VARCHAR(255) NOT NULL COMMENT '유저 비밀번호',
    USER_STATUS VARCHAR(10) NOT NULL COMMENT '유저 상태',
    USER_AUTH VARCHAR(10) NOT NULL COMMENT '유저 권한',
    PET_CODE INT NOT NULL COMMENT '애완견 코드',
    REGIST_DATE DATE NOT NULL COMMENT '유저 등록 날짜',
    UPDATE_DATE DATE NOT NULL COMMENT '유저 수정 날짜',
    CONSTRAINT PK_USER_CODE PRIMARY KEY (USER_CODE)
) ENGINE=INNODB COMMENT '유저 테이블';

CREATE TABLE IF NOT EXISTS SUSPENSION_REASON
(
    USER_SUS_CODE INT AUTO_INCREMENT COMMENT '유저 정지 코드',
    USER_CODE INT NOT NULL COMMENT '유저 코드',
    SUS_REASON VARCHAR(255) NOT NULL COMMENT '정지 사유',
    SUS_DAYS INT NOT NULL COMMENT '유저 정지 일수',
    REGIST_DATE DATE NOT NULL COMMENT '정지 등록 일자',
    CONSTRAINT PK_USER_SUS_CODE PRIMARY KEY (USER_SUS_CODE)
)ENGINE=INNODB COMMENT '정지 사유 테이블';

CREATE TABLE IF NOT EXISTS PET_TABLE
(
    PET_CODE INT AUTO_INCREMENT COMMENT '애완견 코드',
    PET_NAME VARCHAR(20) NOT NULL COMMENT '애완견 이름',
    PET_STATUS VARCHAR(10) NOT NULL COMMENT '애완견 상태',
    PET_SPECIES VARCHAR(20) NOT NULL COMMENT '애완견 종',
    PET_WEIGHT VARCHAR(5) NOT NULL COMMENT '애완견 무게',
    REGIST_DATE DATE NOT NULL COMMENT '펫 등록 날짜',
    UPDATE_DATE DATE NOT NULL COMMENT '펫 수정 날짜',
    CONSTRAINT PK_PET_CODE PRIMARY KEY (PET_CODE)
) ENGINE=INNODB COMMENT '애완견 테이블';

CREATE TABLE IF NOT EXISTS PET_IMG_TABLE
(
    PET_IMG_CODE INT AUTO_INCREMENT COMMENT '애완견 이미지 코드',
    PET_CODE INT NOT NULL COMMENT '애완견 코드',
    IMG_ORG_NAME VARCHAR(255) NOT NULL COMMENT '이미지 원본 이름',
    IMG_UPD_NAME VARCHAR(255) NOT NULL COMMENT '이미지 업데이트 이름',
    IMG_STATUS VARCHAR(10) NOT NULL COMMENT '이미지 상태',
    CONSTRAINT PK_PET_IMG_CODE PRIMARY KEY (PET_IMG_CODE)
)ENGINE=INNODB COMMENT '애완견 사진 테이블';

CREATE TABLE IF NOT EXISTS DOG_OWNER_MAPPING
(
    USER_CODE INT NOT NULL COMMENT '유저 코드',
    PET_CODE INT NOT NULL COMMENT '애완견 코드',
    CONSTRAINT DOG_OWN_FK_USER_CODE FOREIGN KEY (USER_CODE) REFERENCES USER_TABLE (USER_CODE),
    CONSTRAINT DOG_OWN_FK_PET_CODE FOREIGN KEY (PET_CODE) REFERENCES PET_TABLE (PET_CODE)
)ENGINE=INNODB COMMENT '견주 매핑 테이블';

CREATE TABLE IF NOT EXISTS ANNOUNCEMENT
(
    ANNO_CODE INT AUTO_INCREMENT COMMENT '공지사항 코드',
    USER_CODE INT NOT NULL COMMENT '유저 코드',
    ANNO_TITLE VARCHAR(20) NOT NULL COMMENT '공지사항 제목',
    ANNO_CONTENTS VARCHAR(255) NOT NULL COMMENT '공지사항 내용',
    ANNO_PIN INT NOT NULL DEFAULT 0 COMMENT '상단 고정 여부',
    ANNO_STATUS VARCHAR(20) NOT NULL COMMENT '공지글 상태',
    REGIST_DATE DATE NOT NULL COMMENT '공지글 등록 날짜',
    UPDATE_DATE DATE NOT NULL COMMENT '공지글 수정 날짜',
    CONSTRAINT PK_ANNO_CODE PRIMARY KEY (ANNO_CODE),
    CONSTRAINT ANNO_FK_USER_CODE FOREIGN KEY (USER_CODE) REFERENCES USER_TABLE (USER_CODE)
) ENGINE=INNODB COMMENT '공지사항 테이블';

CREATE TABLE IF NOT EXISTS INQUIRY
(
    INQ_CODE INT AUTO_INCREMENT COMMENT '문의 코드',
    USER_CODE INT NOT NULL COMMENT '유저 코드',
    INQ_TITLE VARCHAR(20) NOT NULL COMMENT '문의 제목',
    INQ_CLASSI VARCHAR(5) NOT NULL COMMENT '문의 분류',
    INQ_CONTENTS VARCHAR(255) NOT NULL COMMENT '문의 내용',
    REGIST_DATE DATE NOT NULL COMMENT '공지글 등록 날짜',
    UPDATE_DATE DATE NOT NULL COMMENT '공지글 수정 날짜',
    CONSTRAINT PK_INQ_CODE PRIMARY KEY (INQ_CODE),
    CONSTRAINT INQ_FK_USER_CODE FOREIGN KEY (USER_CODE) REFERENCES USER_TABLE (USER_CODE)
)ENGINE=INNODB COMMENT '문의 테이블';

CREATE TABLE IF NOT EXISTS INQUIRY_ANSWER
(
    ANSWER_CODE INT AUTO_INCREMENT COMMENT '문의 답변 코드',
    INQ_CODE INT NOT NULL COMMENT '문의 코드',
    ANSWER_CONTENTS VARCHAR(255) NOT NULL COMMENT '문의 답변 내용',
    REGIST_DATE DATE NOT NULL COMMENT '공지글 등록 날짜',
    UPDATE_DATE DATE NOT NULL COMMENT '공지글 수정 날짜',
    CONSTRAINT PK_ANS_CODE PRIMARY KEY (ANSWER_CODE),
    CONSTRAINT INQ_ANS_FK_INQ_CODE FOREIGN KEY (INQ_CODE) REFERENCES INQUIRY (INQ_CODE)
)ENGINE=INNODB COMMENT '문의 답변 테이블';

CREATE TABLE IF NOT EXISTS INQUIRY_IMAGE
(
    INQUIRY_IMG_CODE INT AUTO_INCREMENT COMMENT '문의 이미지 코드',
    INQ_CODE INT NOT NULL COMMENT '문의 코드',
    IMG_ORG_NAME VARCHAR(255) NOT NULL COMMENT '문의 이미지 원본 이름',
    IMG_UPD_NAME VARCHAR(255) NOT NULL COMMENT '이미지 수정된 이름',
    REGIST_DATE DATE NOT NULL COMMENT '문의 이미지 등록 시간',
    CONSTRAINT PK_INQUIRY_IMG_CODE PRIMARY KEY (INQUIRY_IMG_CODE)
)ENGINE=INNODB COMMENT '문의 이미지 테이블';

CREATE TABLE IF NOT EXISTS COMPANY_TABLE
(
    COMPANY_CODE INT AUTO_INCREMENT COMMENT '업체 코드',
    COMPANY_CLASSI VARCHAR(10) NOT NULL COMMENT '업체 분류',
    COMPANY_STATUS VARCHAR(10) NOT NULL COMMENT '업체 상태',
    COMPANY_NAME VARCHAR(20) NOT NULL COMMENT '업체 이름',
    COMPANY_ADDRESS VARCHAR(70) NOT NULL COMMENT '업체 주소',
    COMPANY_LAT VARCHAR(255) NOT NULL COMMENT '업체 위도',
    COMAPNY_HAR VARCHAR(255) NOT NULL COMMENT '업체 경도',
    COMPANY_PHONE_NUMBER VARCHAR(30) NOT NULL COMMENT '업체 번호',
    COMPANY_OPEN_TIME VARCHAR(10) NOT NULL COMMENT '오픈 시간',
    COMPANY_CLOSE_TIME VARCHAR(10) NOT NULL COMMENT '마감 시간',
    REGIST_DATE DATE NOT NULL COMMENT '업체 등록 날짜',
    UPDATE_DATE DATE NOT NULL COMMENT '업체 수정 날짜',
    CONSTRAINT PK_COMPANY_CODE PRIMARY KEY (COMPANY_CODE)
)ENGINE=INNODB COMMENT '업체 테이블';

CREATE TABLE IF NOT EXISTS TOUR_PLAN_TABLE
(
    TOUR_PLAN_CODE INT AUTO_INCREMENT COMMENT '여행 계획 코드',
    COMPANY_CODE INT NOT NULL COMMENT '업체 코드',
    TOUR_TITLE VARCHAR(20) NOT NULL COMMENT '계획 제목',
    TOUR_CONTENTS VARCHAR(255) NOT NULL COMMENT '여행 설명',
    TOUR_STATUS VARCHAR(10) NOT NULL COMMENT '여행 상태',
    TOUR_DAYS INT NOT NULL COMMENT '여행 일 수',
    REGIST_DATE DATE NOT NULL COMMENT '여행 등록 날짜',
    UPDATE_DATE DATE NOT NULL COMMENT '여행 수정 날짜',
    CONSTRAINT PK_TOUR_PLAN_CODE PRIMARY KEY (TOUR_PLAN_CODE)
)ENGINE=INNODB COMMENT '여행 계획 테이블';

CREATE TABLE IF NOT EXISTS SELECT_PLAN
(
    SEL_PLAN_CODE INT AUTO_INCREMENT COMMENT '선택 계획 코드',
    USER_CODE INT NOT NULL COMMENT '유저 코드',
    SEL_PLAN_STATUS VARCHAR(10) NOT NULL COMMENT '계획 상태',
    TOUR_PLAN_CODE INT NOT NULL COMMENT '여행 계획 코드',
    START_DAY DATE NOT NULL COMMENT '여행 시작 날',
    END_DAY DATE NOT NULL COMMENT '여행 마지막 날',
    REGIST_DATE DATE NOT NULL COMMENT '여행 등록 날짜',
    UPDATE_DATE DATE NOT NULL COMMENT '여행 수정 날짜',
    CONSTRAINT PK_SEL_PLAN_CODE PRIMARY KEY (SEL_PLAN_CODE),
    CONSTRAINT USER_TABLE_FK_USER_CODE FOREIGN KEY (USER_CODE) REFERENCES USER_TABLE (USER_CODE),
    CONSTRAINT TOUR_PLAN_TABLE_FK_TOUR_PLAN_CODE FOREIGN KEY (TOUR_PLAN_CODE) REFERENCES TOUR_PLAN_TABLE (TOUR_PLAN_CODE)
)ENGINE=INNODB COMMENT '선택 계획 테이블';

CREATE TABLE IF NOT EXISTS COMPANY_IMG
(
    COM_IMG_CODE INT AUTO_INCREMENT COMMENT '업체 이미지 코드',
    COMPANY_CODE INT NOT NULL COMMENT '업체 코드',
    COM_IMG_ORG VARCHAR(255) NOT NULL COMMENT '업체 이미지 원본 이름',
    COM_IMG_UPD VARCHAR(255) NOT NULL COMMENT '업체 이미지 수정 이름',
    COM_IMG_STATUS VARCHAR(10) NOT NULL COMMENT '업체 이미지 상태',
    REGIST_DATE DATE NOT NULL COMMENT '업체 이미지 등록 시간',
    CONSTRAINT PK_COM_IMG_CODE PRIMARY KEY (COM_IMG_CODE),
    CONSTRAINT COMPANY_TABLE_FK_COMPANY_CODE FOREIGN KEY (COMPANY_CODE) REFERENCES COMPANY_TABLE (COMPANY_CODE)
)ENGINE=INNODB COMMENT '업체 이미지 테이블';

CREATE TABLE IF NOT EXISTS COMPANY_HOLIDAY
(
    HOLIDAY_CODE INT AUTO_INCREMENT COMMENT '휴무일 코드',
    COMPANY_CODE INT NOT NULL COMMENT '업체 코드',
    HOLIDAY_STATUS VARCHAR(10) NOT NULL COMMENT '휴무일 상태',
    HOLIDAY_WHETHER VARCHAR(5) NOT NULL COMMENT '휴무일 유무',
    DESIGNATED_HOLIDAY VARCHAR(10) COMMENT '지정 휴무일',
    SPECIAL_HOLIDAY VARCHAR(10) COMMENT '특수 휴무일',
    CONSTRAINT PK_HOLIDAY_CODE PRIMARY KEY(HOLIDAY_CODE),
    CONSTRAINT COMPANY_HOLIDAY_TABLE_FK_COMPANY_CODE FOREIGN KEY (COMPANY_CODE) REFERENCES COMPANY_TABLE (COMPANY_CODE)
)ENGINE=INNODB COMMENT '업체 휴무일 테이블';

CREATE TABLE IF NOT EXISTS COMPANY_RESTRICTIONS
(
    RESTRIC_CODE INT AUTO_INCREMENT COMMENT '체크사항 코드',
    COMPANY_CODE INT NOT NULL COMMENT '업체 코드',
    POS_PET_WEIGHT VARCHAR(5) NOT NULL COMMENT '입장 가능한 동물의 무게',
    PET_RESTRIC VARCHAR(20) NOT NULL COMMENT '입장 가능 제한 사항',
    PET_ACCOM_IN VARCHAR(10) NOT NULL COMMENT '실내 동반 여부',
    PET_ACCOM_OUT VARCHAR(10) NOT NULL COMMENT '실외 동반 여부',
    CONSTRAINT PK_RESTRIC_CODE PRIMARY KEY (RESTRIC_CODE),
    CONSTRAINT COMPANY_COMPANY_RESTRICTIONS_TABLE_FK_COMPANY_CODE FOREIGN KEY (COMPANY_CODE) REFERENCES COMPANY_TABLE (COMPANY_CODE)
) ENGINE=INNODB COMMENT '업체 체크사항 테이블';

CREATE TABLE IF NOT EXISTS LIKE_TABLE
(
    LIKE_CODE INT AUTO_INCREMENT COMMENT '좋아요 코드',
    COMPANY_CODE INT NOT NULL COMMENT '업체 코드',
    USER_CODE INT NOT NULL COMMENT '유저 코드',
    REGIST_DATE DATE NOT NULL COMMENT '좋아요 등록 날짜',
    CONSTRAINT PK_LIKE_CODE PRIMARY KEY (LIKE_CODE),
    CONSTRAINT LIKE_TABLE_FK_COMPANY_CODE FOREIGN KEY (COMPANY_CODE) REFERENCES COMPANY_TABLE (COMPANY_CODE),
    CONSTRAINT LIKE_TABLE_FK_USER_CODE FOREIGN KEY (USER_CODE) REFERENCES USER_TABLE (USER_CODE)
)ENGINE=INNODB COMMENT '좋아요 테이블';

CREATE TABLE IF NOT EXISTS COMPANY_REVIEW
(
    REVIEW_CODE INT AUTO_INCREMENT COMMENT '리뷰 코드',
    COMPANY_CODE INT NOT NULL COMMENT '업체 코드',
    USER_CODE INT NOT NULL COMMENT '유저 코드',
    COMPANY_GRADE INT NOT NULL COMMENT '업체 평점',
    REVIEW_CONTENTS VARCHAR(255) NOT NULL COMMENT '리뷰 내용',
    REVIEW_STATUS VARCHAR(10) NOT NULL COMMENT '리뷰 상태',
    REGIST_DATE DATE NOT NULL COMMENT '리뷰 등록 날짜',
    UPDATE_DATE DATE NOT NULL COMMENT '리뷰 수정 날짜',
    CONSTRAINT PK_REVIEW_CODE PRIMARY KEY (REVIEW_CODE),
    CONSTRAINT REVIEW_TABLE_FK_COMPANY_CODE FOREIGN KEY (COMPANY_CODE) REFERENCES COMPANY_TABLE (COMPANY_CODE),
    CONSTRAINT REVIEW_TABLE_FK_USER_CODE FOREIGN KEY (USER_CODE) REFERENCES USER_TABLE (USER_CODE)
)ENGINE=INNODB COMMENT '업체 리뷰 테이블';

CREATE TABLE IF NOT EXISTS REVIEW_IMG
(
    REV_IMG_CODE INT AUTO_INCREMENT COMMENT '리뷰 사진 코드',
    REV_IMG_ORG VARCHAR(255) NOT NULL COMMENT '리뷰 이미지 원본 이름',
    REV_IMG_UPD VARCHAR(255) NOT NULL COMMENT '리뷰 이미지 수정 이름',
    REVIEW_CODE INT NOT NULL COMMENT '리뷰 코드',
    REV_IMG_STATUS VARCHAR(10) NOT NULL COMMENT'리뷰 사진 상태',
    REGIST_DATE DATE NOT NULL COMMENT '리뷰 사진 등록 날짜',
    UPDATE_DATE DATE NOT NULL COMMENT '리뷰 사진 수정 날짜',
    CONSTRAINT PK_REV_IMG_CODE PRIMARY KEY (REV_IMG_CODE),
    CONSTRAINT REVIEW_IMG_FK_REVIEW_CODE FOREIGN KEY (REVIEW_CODE) REFERENCES COMPANY_REVIEW (REVIEW_CODE)
)ENGINE=INNODB COMMENT '리뷰 이미지 테이블';

CREATE TABLE IF NOT EXISTS COMPANY_OFF_APP
(
    COMPANY_OFF_CODE INT AUTO_INCREMENT COMMENT '업체 관계자 코드',
    USER_CODE INT NOT NULL COMMENT '유저 코드',
    COMPANY_CODE INT NOT NULL COMMENT '업체 코드',
    BUSINESS_APP VARCHAR(5) NOT NULL COMMENT '사업자 등록 승인 여부',
    APP_STATUS VARCHAR(5) NOT NULL COMMENT '승인 상태',
    NOT_APP_REASON VARCHAR(255) NOT NULL COMMENT '미승인 사유',
    CONSTRAINT PK_COMPANY_OFF_CODE PRIMARY KEY (COMPANY_OFF_CODE),
    CONSTRAINT COMPANY_OFF_APP_FK_USER_CODE FOREIGN KEY(USER_CODE) REFERENCES USER_TABLE (USER_CODE),
    CONSTRAINT COMPANY_OFF_APP_FK_COMPANY_CODE FOREIGN KEY(COMPANY_CODE) REFERENCES COMPANY_TABLE (COMPANY_CODE)
)ENGINE=INNODB COMMENT '업체 관계자 승인 테이블';
